# checkpoint-restart related settings

# can be set to either true/false
# (default: true)
#
# enable/disable the CR feature
#
checkpoint-enabled true

# can be set to either file/advert/none
# (default: file)
#
# restore file dump method:
# file -> local filesystem read/write
# advert -> remote advertdb (not implemented)
#
checkpoint-method file

# can be set to either true/false
# (default: true)
#
# if compression is enabled, then the restore
# file contents are compressed before written
# to permanent storage.
#
checkpoint-compress true

# can be set to either true/false
# (default: true)
#
# if hashlog is enabled, then the restore file
# name is unique to the input DAG. If disabled
# the restore file name is set to "digedag-CR.log"
# and the file output directory is set to "."
# (current working directory)
#
checkpoint-hashlog true

# can be set to either true/false
# (default: false)
#
# if enabled, *really* verbose debug output
# is produced at various points of the
# checkpointing processes.
#
checkpoint-verbose false

# points to path/file for local disk method
# (default: "/tmp/")
#
# sets the restore file output directory
#
checkpoint-filepath /tmp/
